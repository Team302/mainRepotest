230 void SwerveModule::SetDesiredState
231 (
232  const SwerveModuleState& targetState
233 )
234 {
235  // Update targets so the angle turned is less than 90 degrees
236  // If the desired angle is less than 90 degrees from the target angle (e.g., -90 to 90 is the amount of turn), just use the angle and speed values
237  // if it is more than 90 degrees (90 to 270), the can turn the opposite direction -- increase the angle by 180 degrees -- and negate the wheel speed
238  // finally, get the value between -90 and 90
239  Rotation2d currAngle = Rotation2d(units::angle::degree_t(m_turnSensor.get()->GetAbsolutePosition()));
240  auto optimizedState = Optimize(targetState, currAngle);
241  // auto optimizedState = SwerveModuleState::Optimize(targetState, currAngle);
242  // auto optimizedState = targetState;
243 
244  // Set Turn Target
245  SetTurnAngle(optimizedState.angle.Degrees());
246 
247  // Set Drive Target
248  SetDriveSpeed(optimizedState.speed);
249 }

251 // Note: the following was taken from the WPI code and tweaked because we were seeing some weird
252 // reversals that we believe was due to not using a tolerance
253 SwerveModuleState SwerveModule::Optimize
254 (
255  const SwerveModuleState& desiredState,
256  const Rotation2d& currentAngle
257 )
258 {
259 
260  auto delta = desiredState.angle - currentAngle;
261 
262  Logger::GetLogger()->ToNtTable("Optimize", "current", currentAngle.Degrees().to<double>());
263  Logger::GetLogger()->ToNtTable("Optimize", "target", desiredState.angle.Degrees().to<double>());
264  Logger::GetLogger()->ToNtTable("Optimize", "delta", delta.Degrees().to<double>());
278  // make sure the delta is between -180.0 and 180.0
279  if (delta.Degrees() < -180.0_deg)
280  {
281  while (delta.Degrees() < -180.0_deg )
282  {
283  delta.Degrees() = delta.Degrees() + 360.0_deg;
284  }
285  }
286  else if (delta.Degrees() > 180.0_deg)
287  {
288  while (delta.Degrees() > 180.0_deg)
289  {
290  delta.Degrees() = delta.Degrees() - 360.0_deg;
291  }
292  }
293 
294  // if the delta is > 90 degrees, rotate the
295 // if ((units::math::abs(delta.Degrees()) - 90_deg) > 0.1_deg)
296  if ((units::math::abs(delta.Degrees()) - 160_deg) > 0.1_deg)
297  {
298  Logger::GetLogger()->ToNtTable("Optimize", "optimized", (desiredState.angle + Rotation2d{180_deg}).Degrees().to<double>());
299  return {-desiredState.speed, desiredState.angle + Rotation2d{180_deg}};
300  }
301  else
302  {
303  Logger::GetLogger()->ToNtTable("Optimize", "optimized", desiredState.angle.Degrees().to<double>());
304  return {desiredState.speed, desiredState.angle};
305  }
306 }

320 void SwerveModule::SetDriveSpeed( units::velocity::meters_per_second_t speed )
321 {
322  if ( abs(speed.to<double>()/m_maxVelocity.to<double>()) < 0.05 )
323  {
324  m_currentState.speed = 0_mps;
325  }
326  else
327  {
328  m_currentState.speed = speed;
329  }
330  Logger::GetLogger()->ToNtTable(m_nt, string("State Speed - mps"), m_currentState.speed.to<double>() );
331  Logger::GetLogger()->ToNtTable(m_nt, string("Wheel Diameter - meters"), units::length::meter_t(m_wheelDiameter).to<double>() );
332 
333  if (m_runClosedLoopDrive)
334  {
335  // convert mps to unitless rps by taking the speed and dividing by the circumference of the wheel
336  auto driveTarget = m_currentState.speed.to<double>() / (units::length::meter_t(m_wheelDiameter).to<double>() * wpi::math::pi);
337  driveTarget /= m_driveMotor.get()->GetGearRatio();
338  driveTarget *= m_scale;
339 
340  Logger::GetLogger()->ToNtTable(m_nt, string("drive motor id"), m_driveMotor.get()->GetID() );
341  Logger::GetLogger()->ToNtTable(m_nt, string("drive target"), driveTarget );
342  Logger::GetLogger()->ToNtTable(m_nt, string("drive scale"), m_scale );
343 
344  m_driveMotor.get()->SetControlMode(ControlModes::CONTROL_TYPE::VELOCITY_RPS);
345  m_driveMotor.get()->Set(m_nt, driveTarget);
346  }
347  else
348  {
349  auto percent = m_currentState.speed / m_maxVelocity;
350  percent *= m_scale;
351  m_driveMotor.get()->SetControlMode(ControlModes::CONTROL_TYPE::PERCENT_OUTPUT);
352  m_driveMotor.get()->Set(m_nt, percent);
353  }
354 }

356 void SwerveModule::SetTurnAngle( units::angle::degree_t targetAngle )
357 {
358  m_currentState.angle = targetAngle;
359  Rotation2d currAngle = Rotation2d(units::angle::degree_t(m_turnSensor.get()->GetAbsolutePosition()));
360  Rotation2d deltaAngle = targetAngle - currAngle.Degrees();
361 
362  Logger::GetLogger()->ToNtTable(m_nt, string("turn motor id"), m_turnMotor.get()->GetID() );
363  Logger::GetLogger()->ToNtTable(m_nt, string("current angle"), currAngle.Degrees().to<double>() );
364  Logger::GetLogger()->ToNtTable(m_nt, string("target angle"), targetAngle.to<double>() );
365  Logger::GetLogger()->ToNtTable(m_nt, string("delta angle"), deltaAngle.Degrees().to<double>() );
366 
367  if ( abs(deltaAngle.Degrees().to<double>()) > 1.0 )
368  {
369  auto motor = m_turnMotor.get()->GetSpeedController();
370  auto fx = dynamic_cast<WPI_TalonFX*>(motor.get());
371  auto sensors = fx->GetSensorCollection();
372  //=============================================================================
373  // 5592 counts on the falcon for 76.729 degree change on the CANCoder (wheel)
374  //=============================================================================
375  double deltaTicks = (deltaAngle.Degrees().to<double>() * 5592 / 76.729);
376  double currentTicks = sensors.GetIntegratedSensorPosition();
377  double desiredTicks = currentTicks + deltaTicks;
378 
379  Logger::GetLogger()->ToNtTable(m_nt, string("currentTicks"), currentTicks );
380  Logger::GetLogger()->ToNtTable(m_nt, string("deltaTicks"), deltaTicks );
381  Logger::GetLogger()->ToNtTable(m_nt, string("desiredTicks"), desiredTicks );
382 
383  m_turnMotor.get()->SetControlMode(ControlModes::CONTROL_TYPE::POSITION_ABSOLUTE);
384  m_turnMotor.get()->Set(m_nt, desiredTicks);
385  }
386  else
387  {
388  m_turnMotor.get()->SetControlMode(ControlModes::CONTROL_TYPE::PERCENT_OUTPUT);
389  m_turnMotor.get()->Set(m_nt, 0.0);
390  }
