198 shared_ptr<IDragonMotorController> SwerveChassisFactory::GetMotorController
199 (
200  const IDragonMotorControllerMap& motorControllers,
201  MotorControllerUsage::MOTOR_CONTROLLER_USAGE usage
202 )
203 {
204  shared_ptr<IDragonMotorController> motor;
205  auto it = motorControllers.find( usage );
206  if ( it != motorControllers.end() ) // found it
207  {
208  motor = it->second;
209  }
210  else
211  {
212  string msg = "motor not found; usage = ";
213  msg += to_string( usage );
214  Logger::GetLogger()->LogError( string( "SwerveChassisFactory::GetMotorController" ), msg );
215  }
216 
217  if ( motor.get() == nullptr )
218  {
219  string msg = "motor is nullptr; usage = ";
220  msg += to_string( usage );
221  Logger::GetLogger()->LogError( string( "SwerveChassisFactory::GetMotorController" ), msg );
222  }
223  return motor;
224 }
