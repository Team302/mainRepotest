108 void Robot::TeleopInit()
109 {
110  m_drive = make_shared<SwerveDrive>();
111  m_drive.get()->Init();
112 
113  auto shooter = MechanismFactory::GetMechanismFactory()->GetShooter();
114  m_shooterState = ( shooter.get() != nullptr ) ? ShooterStateMgr::GetInstance() : nullptr;
115  if ( m_shooterState != nullptr )
116  {
117  m_shooterState->SetCurrentState(ShooterStateMgr::SHOOTER_STATE::OFF, true);
118  }
119  else
120  {
121  Logger::GetLogger()->LogError(Logger::LOGGER_LEVEL::ERROR_ONCE, string("TeleopInit"), string("no shooter state manager"));
122  }
123 
124  //set camera to drivermode to stream to dashboard
125  if ( m_limelight != nullptr)
126  {
127  m_driverMode->SetCamToDriveMode( m_limelight );
128  }
