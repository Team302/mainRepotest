158 bool Mech2MotorState::AtTarget() const
159 {
160  auto same = true;
161  if ( m_mechanism != nullptr )
162  {
163  if ( m_positionBased && !m_speedBased )
164  {
165  same = ( abs( m_primaryTarget - m_mechanism->GetPrimaryPosition()) < 1.0 &&
166  abs( m_secondaryTarget - m_mechanism->GetSecondaryPosition()) < 1.0 );
167  }
168  else if ( !m_positionBased && m_speedBased )
169  {
170  same = ( abs( m_primaryTarget - m_mechanism->GetPrimarySpeed()) < 1.0 &&
171  abs( m_secondaryTarget - m_mechanism->GetSecondarySpeed()) < 1.0 );
172  }
173  else if ( m_positionBased && m_speedBased )
174  {
175  same = ( ( abs( m_primaryTarget - m_mechanism->GetPrimaryPosition()) < 1.0 &&
176  abs( m_secondaryTarget - m_mechanism->GetSecondaryPosition()) < 1.0 ) ||
177  ( abs( m_primaryTarget - m_mechanism->GetPrimarySpeed()) < 1.0 &&
178  abs( m_secondaryTarget - m_mechanism->GetSecondarySpeed()) < 1.0 ) );
179  }
180  }
181  return same;
182 }
