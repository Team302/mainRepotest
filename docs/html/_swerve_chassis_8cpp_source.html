113  Logger::GetLogger()->ToNtTable("Swerve Chassis", "XSpeed", xSpeed.to<double>() );

115  Logger::GetLogger()->ToNtTable("Swerve Chassis", "ZSpeed", rot.to<double>() );

118 
119  if ( (abs(xSpeed.to<double>()) < m_deadband) &&

122  {
123  m_frontLeft.get()->StopMotors();
124  m_frontRight.get()->StopMotors();

136 
137  //auto states = m_kinematics.ToSwerveModuleStates(ChassisSpeeds{xSpeed, ySpeed, rot} );
138 
139  m_kinematics.NormalizeWheelSpeeds(&states, m_maxSpeed);

141  auto [fl, fr, bl, br] = states;

172 void SwerveChassis::Drive( double drive, double steer, double rotate, bool fieldRelative )
173 {
174  if ( abs(drive) < m_deadband &&
175  abs(steer) < m_deadband &&
176  abs(rotate) < m_deadband )
177  {
178  // feed the motors
179  m_frontLeft.get()->StopMotors();
180  m_frontRight.get()->StopMotors();
181  m_backLeft.get()->StopMotors();
182  m_backRight.get()->StopMotors();
183  }
184  else
185  {
186  // scale joystick values to velocities using max chassis values
187  auto maxSpeed = GetMaxSpeed();
188  auto maxRotation = GetMaxAngularSpeed();
189 
190  Logger::GetLogger()->ToNtTable("Swerve Chassis", "MaxSpeed", maxSpeed.to<double>() );
191  Logger::GetLogger()->ToNtTable("Swerve Chassis", "maxRotation", maxRotation.to<double>() );
192 
193  units::velocity::meters_per_second_t driveSpeed = drive * maxSpeed;
194  units::velocity::meters_per_second_t steerSpeed = steer * maxSpeed;
195  units::angular_velocity::radians_per_second_t rotateSpeed = rotate * maxRotation;
196 
197  Drive( driveSpeed, steerSpeed, rotateSpeed, fieldRelative );
198  }
199 }
