75 units::angle::degree_t DragonLimelight::GetTargetHorizontalOffset() const
76 {
77  units::angle::degree_t tx = units::angle::degree_t(m_networktable.get()->GetNumber("tx", 0.0));
78  units::angle::degree_t ty = units::angle::degree_t(m_networktable.get()->GetNumber("ty", 0.0));
79  if(m_rotation == units::angle::degree_t(0.0))
80  {
81  return tx;
82  }
83  else if(m_rotation == units::angle::degree_t(90.0))
84  {
85  return -ty;
86  }
87  else if(m_rotation == units::angle::degree_t(180.0))
88  {
89  return -tx;
90  }
91  else if(m_rotation == units::angle::degree_t(270.0))
92  {
93  return ty;
94  }
95  else
96  {
97  Logger::GetLogger()->LogError("DragonLimelight::GetTargetVerticalOffset", "Invalid limelight rotation");
98  return units::angle::degree_t(-180.0);
99  }
