57 ShooterStateMgr::ShooterStateMgr() : m_states(),
58  m_currentState()
59 {
60  // Parse the configuration file
61  auto stateXML = make_unique<StateDataDefn>();
62  vector<MechanismTargetData*> targetData = stateXML.get()->ParseXML( MechanismTypes::MECHANISM_TYPE::SHOOTER );
63 
64  // initialize the xml string to state map
65  map<string, SHOOTER_STATE> stateStringToEnumMap;
66  stateStringToEnumMap["SHOOTEROFF"] = SHOOTER_STATE::OFF;
67  stateStringToEnumMap["SHOOTERGETREADY"] = SHOOTER_STATE::GET_READY;
68  stateStringToEnumMap["SHOOTERSHOOTGREEN"] = SHOOTER_STATE::SHOOTGREEN;
69  stateStringToEnumMap["SHOOTERSHOOTYELLOW"] = SHOOTER_STATE::SHOOTYELLOW;
70  stateStringToEnumMap["SHOOTERSHOOTBLUE"] = SHOOTER_STATE::SHOOTBLUE;
71  stateStringToEnumMap["SHOOTERSHOOTRED"] = SHOOTER_STATE::SHOOTRED;
72 
73  auto ntName = MechanismFactory::GetMechanismFactory()->GetShooter().get()->GetNetworkTableName();
74  Logger::GetLogger()->ToNtTable(ntName, "SHOOTEROFF", "not created");
75  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERGETREADY", "not created");
76  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTGREEN", "not created");
77  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTYELLOW", "not created");
78  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTBLUE", "not created");
79  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTRED", "not created");
80  for (auto inx=0; inx<MAX_SHOOTER_STATES; ++inx)
81  {
82  m_states[inx] = nullptr;
83  }
84 
85  // create the states passing the configuration data
86  for ( auto td: targetData )
87  {
88  auto stateString = td->GetStateString();
89  auto stateStringToEnumMapItr = stateStringToEnumMap.find( stateString );
90  if ( stateStringToEnumMapItr != stateStringToEnumMap.end() )
91  {
92  auto stateEnum = stateStringToEnumMapItr->second;
93  if ( m_states[stateEnum] == nullptr )
94  {
95  auto controlData = td->GetController();
96  auto controlData2 = td->GetController2();
97  auto target1 = td->GetTarget();
98  auto target2 = td->GetSecondTarget();
99 
100  switch ( stateEnum )
101  {
102  case SHOOTER_STATE::OFF:
103  {
104  m_states[SHOOTER_STATE::OFF] = new ShooterState( controlData, controlData2, target1, target2 );
105  m_currentState = m_states[SHOOTER_STATE::OFF];
106  m_currentStateEnum = stateEnum;
107  m_currentState->Init();
108  Logger::GetLogger()->ToNtTable(ntName, "SHOOTEROFF", "created");
109  }
110  break;
111 
112  case SHOOTER_STATE::GET_READY:
113  {
114  m_states[stateEnum] = new ShooterState( controlData, controlData2, target1, target2 );
115  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERGETREADY", "created");
116  }
117  break;
118 
119  case SHOOTER_STATE::SHOOTGREEN:
120  {
121  m_states[stateEnum] = new ShooterState( controlData, controlData2, target1, target2 );
122  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTGREEN", "created");
123  }
124  break;
125 
126  case SHOOTER_STATE::SHOOTYELLOW:
127  {
128  m_states[stateEnum] = new ShooterState( controlData, controlData2, target1, target2 );
129  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTYELLOW", "created");
130  }
131  break;
132 
133  case SHOOTER_STATE::SHOOTBLUE:
134  {
135  m_states[stateEnum] = new ShooterState( controlData, controlData2, target1, target2 );
136  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTBLUE", "created");
137  }
138  break;
139 
140  case SHOOTER_STATE::SHOOTRED:
141  {
142  m_states[stateEnum] = new ShooterState( controlData, controlData2, target1, target2 );
143  Logger::GetLogger()->ToNtTable(ntName, "SHOOTERSHOOTRED", "created");
144  }
145  break;
146 
147  default:
148  {
149  Logger::GetLogger()->LogError( string("ShooterStateMgr::ShooterStateMgr"), string("unknown state"));
150  }
151  break;
152  }
153  }
154  else
155  {
156  Logger::GetLogger()->LogError( string("ShooterStateMgr::ShooterStateMgr"), string("multiple mechanism state info for state"));
157  }
158  }
159  else
160  {
161  Logger::GetLogger()->LogError( string("ShooterStateMgr::ShooterStateMgr"), string("state not found"));
162  }
163  }
164 }

168 void ShooterStateMgr::RunCurrentState()
169 {
170  auto ntName = MechanismFactory::GetMechanismFactory()->GetShooter().get()->GetNetworkTableName();
171  auto controller = TeleopControl::GetInstance();
172 
173  if (controller != nullptr)
174  {
175  if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOTER_PREPARE_TO_SHOOT ))
176  {
177  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Prepare to Shoot");
178  SetCurrentState( SHOOTER_STATE::GET_READY, false );
179  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::HOLD, false);
180  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
181  }
182  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOTER_MANUAL_SHOOT_GREEN ))
183  {
184  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Green");
185  SetCurrentState( SHOOTER_STATE::SHOOTGREEN, false );
186  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::RAPID_RELEASE, false);
187  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
188  }
189  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOTER_MANUAL_SHOOT_YELLOW ))
190  {
191  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Yellow");
192  SetCurrentState( SHOOTER_STATE::SHOOTYELLOW, false );
193  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::RAPID_RELEASE, false);
194  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
195  }
196  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOTER_MANUAL_SHOOT_BLUE ))
197  {
198  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Blue");
199  SetCurrentState( SHOOTER_STATE::SHOOTBLUE, false );
200  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::RAPID_RELEASE, false);
201  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
202  }
203  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOTER_MANUAL_SHOOT_RED ))
204  {
205  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Red");
206  SetCurrentState( SHOOTER_STATE::SHOOTRED, false );
207  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::RAPID_RELEASE, false);
208  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
209  }
210  }
211 
212  // run the current state
213  if ( m_currentState != nullptr )
214  {
215  m_currentState->Run();
216  }
217  BallHopperStateMgr::GetInstance()->RunCurrentState();
218  BallTransferStateMgr::GetInstance()->RunCurrentState();
219 }

223 void ShooterStateMgr::SetCurrentState
224 (
225  SHOOTER_STATE stateEnum,
226  bool run
227 )
228 {
229  auto state = m_states[stateEnum];
230 // if ( state != nullptr && state != m_currentState )
231  if ( state != nullptr )
232  {
233  auto ntName = MechanismFactory::GetMechanismFactory()->GetShooter().get()->GetNetworkTableName();
234  Logger::GetLogger()->ToNtTable(ntName, "Current State", "none");
235 
236  m_currentState = state;
237  m_currentStateEnum = stateEnum;
238  m_currentState->Init();
239  if ( stateEnum == SHOOTER_STATE::GET_READY )
240  {
241  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Prepare to Shoot");
242  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, run );
243  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::HOLD, run);
244  }
245  else if ( stateEnum == SHOOTER_STATE::SHOOTBLUE || stateEnum == SHOOTER_STATE::SHOOTGREEN ||
246  stateEnum == SHOOTER_STATE::SHOOTRED || stateEnum == SHOOTER_STATE::SHOOTYELLOW )
247  {
248  BallTransferStateMgr::GetInstance()->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, run );
249  BallHopperStateMgr::GetInstance()->SetCurrentState( BallHopperStateMgr::RAPID_RELEASE, run);
250  if ( stateEnum == SHOOTER_STATE::SHOOTBLUE)
251  {
252  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Blue");
253  }
254  else if ( stateEnum == SHOOTER_STATE::SHOOTGREEN)
255  {
256  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Green");
257  }
258  else if ( stateEnum == SHOOTER_STATE::SHOOTRED)
259  {
260  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Red");
261  }
262  else
263  {
264  Logger::GetLogger()->ToNtTable(ntName, "Current State", "Shoot Yellow");
265  }
266  }
267  if ( run )
268  {
269  m_currentState->Run();
270  }
271  }
272 }
