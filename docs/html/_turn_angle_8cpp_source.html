54 void TurnAngle::Init
55 (
56  PrimitiveParams* params
57 )
58 {
59  m_maxTime = params->GetTime();
60  m_timer->Reset();
61  m_timer->Start();
62 
63  auto pigeon = PigeonFactory::GetFactory()->GetPigeon();
64  units::degree_t heading(( pigeon != nullptr ) ? pigeon->GetYaw() : 0.0);
65 
66  units::degree_t targetAngle(params->GetHeading());
67 
68  if( params->GetID() == TURN_ANGLE_ABS)
69  {
70  m_targetAngle = targetAngle;
71  } else if( params->GetID() == TURN_ANGLE_REL)
72  {
73  m_targetAngle = targetAngle + heading;
74  } else
75  {
76  Logger::GetLogger()->LogError( string("TurnAngle::Run"), string("turning type not absolute or relative"));
77  }
78 
79  m_driveSpeed = params->GetDriveSpeed();
80 
81  double targetAngleDouble = m_targetAngle.to<double>();
82 
83  //Turning Optimization
84  TurnAngle::IsGreaterThan(targetAngleDouble, m_driveSpeed);
85  TurnAngle::IsLessThan(targetAngleDouble, m_driveSpeed);
86  TurnAngle::ReduceCases(targetAngleDouble, m_driveSpeed);
87 
88  //Create targetAngleDegree which is targetAngleDouble as units::degree_t
89  units::degree_t targetAngleDegree(targetAngleDouble);
90  //Set the targetAngle to the optimized angle
91  m_targetAngle = targetAngleDegree;
92 
93  //Calculate diameter of robot to find arc length
94  double trackSquared = m_chassis.get()->GetTrack().to<double>() * m_chassis.get()->GetTrack().to<double>();
95  double wheelbaseSquared = m_chassis.get()->GetWheelBase().to<double>() * m_chassis.get()->GetWheelBase().to<double>();
96 
97  double hypBeforeSqrt(trackSquared * wheelbaseSquared);
98  units::length::inch_t hypAfterSqrt(sqrt(hypBeforeSqrt));
99 
100  units::length::inch_t m_circumference = pi * hypAfterSqrt;
101 
102  //possiblly get absolute valie after heading - m_targetAngle, if ABS and want to go to 90 and heading is 0, we go -90
103  //does a negative distance mean we go in an opposite direction
104  units::degree_t arcAngle = heading - m_targetAngle;
105 
106  units::length::inch_t m_goalDistance = ( units::math::abs(arcAngle) / units::degree_t(360)) * m_circumference;
107 
108  units::acceleration::meters_per_second_squared_t maxAccel(m_chassis.get()->GetMaxAcceleration());
109  units::meters_per_second_t maxSpeed = m_chassis.get()->GetMaxSpeed();
110 
111  frc::TrapezoidProfile<units::meters>::Constraints m_constraints{ maxSpeed, maxAccel};
112 
113  m_goal = {m_goalDistance, units::velocity::meters_per_second_t( m_driveSpeed) };
114 
115  units::length::meter_t circumferenceMeter(m_circumference);
116 
117  m_distanceToAngleConversion = (2 * pi ) / circumferenceMeter.to<double>();
118 
119 }

121 void TurnAngle::Run()
122 {
123  units::meters_per_second_t xSpeed(0);
124  units::meters_per_second_t ySpeed(0);
125 
126  static constexpr units::second_t kDt = 20_ms;
127 
128  frc::TrapezoidProfile<units::meters> profile{m_constraints, m_goal, m_setpoint};
129 
130  m_setpoint = profile.Calculate(kDt);
131 
132  units::angular_velocity::radians_per_second_t radianTrapezoidSpeed(m_setpoint.velocity.to<double>() * m_distanceToAngleConversion);
133 
134  //Angular speed calculation
135  if (m_reverse)
136  {
137  m_chassis->Drive(xSpeed, ySpeed, -1 * radianTrapezoidSpeed, true);
138  } else if (!m_reverse)
139  {
140  m_chassis->Drive(xSpeed, ySpeed, radianTrapezoidSpeed, true);
141  }
142 }
