193 bool DrivePath::IsSamePose(frc::Pose2d lCurPos, frc::Pose2d lPrevPos)
194 {
195  // Detect if the two poses are the same within a tolerance
196  double dCurPosX = lCurPos.X().to<double>() * 100; //cm
197  double dCurPosY = lCurPos.Y().to<double>() * 100;
198  double dPrevPosX = lPrevPos.X().to<double>() * 100;
199  double dPrevPosY = lPrevPos.Y().to<double>() * 100;
200 
201  int iCurPosX = dCurPosX;
202  int iCurPosY = dCurPosY;
203 
204  int iPrevPosX = dPrevPosX; //remove decimals
205  int iPrevPosY = dPrevPosY;
206 
207  int iDeltaX = abs(iPrevPosX - iCurPosX);
208  int iDeltaY = abs(iPrevPosY - iCurPosY);
209 
210  int iMovThresHold = 1; // cm used for detecting no motion adust this for encoder noise if needed.
211 
212  Logger::GetLogger()->ToNtTable("Deltas", "iDeltaX", to_string(iDeltaX));
213  Logger::GetLogger()->ToNtTable("Deltas", "iDeltaY", to_string(iDeltaY));
214 
215  // If Position of X or Y has moved since last scan.. Using Delta X/Y
216  return (iDeltaX <= iMovThresHold && iDeltaY <= iMovThresHold);
217 }
